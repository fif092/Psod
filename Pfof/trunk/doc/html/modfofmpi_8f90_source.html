37  Subroutine mergehaloes(perco2, mpicomm, periodic)

55  Do f=1,3 ! loop over the faces, we deal with 2 faces at a time

71

73  Call findbridge(nflagloc, nflagrecv, posarr, posava, nbridgearr, bridgearr, nbridgeava, bridgeava, &
74  perco2, 1, 2, mpicomm, periodic)

143  End If bas

145  posava(:,deba) = pos(:,i)
146  strava(deba) = structure(i)
147  deba = deba+1
148  border(i) = border(i) - 8_1
149  End If avant

151  posarr(:,debr) = pos(:,i)
152  strarr(debr) = structure(i)
153  debr = debr+1
154  border(i) = border(i) - 4_1
155  End If arriere
156  droite : If(border(i) >=2 ) Then
157  posdro(:,debd) = pos(:,i)
158  strdro(debd) = structure(i)
159  debd = debd+1
160  border(i) = border(i) - 2_1
161  End If droite
162  gauche : If(border(i) >= 1) Then
163  posgau(:,debg) = pos(:,i)
164  strgau(debg) = structure(i)
165  debg = debg+1
166  End If gauche
167  End If bord
168 
169  End Do bufferface

183  Subroutine findbridge(nloc, nrecv, pos1, pos2, nbridge1, bridge1, nbridge2, bridge2, r2, v1, v2, mpicomm, periodic)
184 
185  Use modmpicom
186  Use modconstant
187  Implicit None
188  ! input
189  Integer(kind=4), intent(in), dimension(6) :: nloc
190  Integer(kind=4), intent(in), dimension(6) :: nrecv
191  Integer(kind=4), intent(in) :: v1

212  ! send flagged positions to direction 2 and receive from direction 1
213  Allocate (posrec(3,nrecv(v1)))
214  Call mpi_irecv(posrec,3*nrecv(v1),mpi_real,neighbours(v1),1,mpicomm,mpireqr1,mpierr)

218  Call mpi_wait(mpireqr1,mpistat,mpierr)

220  ! loop over local flagged positions for direction 1 and received positions from direction 1
221  nbridge1 = 0
222  Do li = 1, nloc(v1)
223  Do ri = 1, nrecv(v1)
224  dx = abs(pos1(1,li)-posrec(1,ri))
225  dy = abs(pos1(2,li)-posrec(2,ri))
226  dz = abs(pos1(3,li)-posrec(3,ri))
227  If(periodic) Then
228  dx = min(dx,1.0-dx)
229  dy = min(dy,1.0-dy)
230  dz = min(dz,1.0-dz)
231  End If
232  d = dx*dx+dy*dy+dz*dz
233  ! count the number of bridges
234  If(d <= r2 ) Then
235  nbridge1 = nbridge1 + 1
236  End If
237  End Do
238  End Do
239 
240  Allocate(bridge1(2,nbridge1))
241 
242  ! we have to loop again to save the local ID of the particles that constitute the bridges
243  ind = 1
244  Do li = 1, nloc(v1)
245  Do ri = 1, nrecv(v1)
246  dx = abs(pos1(1,li)-posrec(1,ri))
247  dy = abs(pos1(2,li)-posrec(2,ri))
248  dz = abs(pos1(3,li)-posrec(3,ri))
249  If(periodic) Then
250  dx = min(dx,1.0-dx)
251  dy = min(dy,1.0-dy)
252  dz = min(dz,1.0-dz)
253  End If
254  d = dx*dx+dy*dy+dz*dz
255  If(d <= r2 ) Then
256  bridge1(1,ind) = li
257  bridge1(2,ind) = ri
258  ind = ind + 1
259  End If
260  End Do
261  End Do
262 
263  Deallocate(posrec)
264 
265 
266  ! send position to direction 1 and receive from direction 2
267  Allocate (posrec(3,nrecv(v2)))
268  Call mpi_irecv(posrec,3*nrecv(v2),mpi_real,neighbours(v2),2,mpicomm,mpireqr1,mpierr)
269  Call mpi_isend(pos1,3*nloc(v1), mpi_real,neighbours(v1),2,mpicomm,mpireqs1,mpierr)
270 
271  Call mpi_wait(mpireqs1,mpistat,mpierr)
272  Call mpi_wait(mpireqr1,mpistat,mpierr)
273 
274  ! same loops with positions corresponding with the second direction
275  nbridge2 = 0
276  Do li = 1, nloc(v2)
277  Do ri = 1, nrecv(v2)
278  dx = abs(pos2(1,li)-posrec(1,ri))
279  dy = abs(pos2(2,li)-posrec(2,ri))
280  dz = abs(pos2(3,li)-posrec(3,ri))
281  If(periodic) Then
282  dx = min(dx,1.0-dx)
283  dy = min(dy,1.0-dy)
284  dz = min(dz,1.0-dz)
285  End If
286  d = dx*dx+dy*dy+dz*dz
287  If(d <= r2 ) Then
288  nbridge2 = nbridge2+1
289  End If
290  End Do
291  End Do
292 
293  Allocate(bridge2(2,nbridge2))
294 
295  ind = 1
296  Do li = 1, nloc(v2)
297  Do ri = 1, nrecv(v2)
298  dx = abs(pos2(1,li)-posrec(1,ri))
299  dy = abs(pos2(2,li)-posrec(2,ri))
300  dz = abs(pos2(3,li)-posrec(3,ri))
301  If(periodic) Then
302  dx = min(dx,1.0-dx)
303  dy = min(dy,1.0-dy)
304  dz = min(dz,1.0-dz)
305  End If
306  d = dx*dx+dy*dy+dz*dz
307  If(d <= r2 ) Then
308  bridge2(1,ind) = li
309  bridge2(2,ind) = ri
310  ind = ind + 1
311  End If
312  End Do
313  End Do
314 
315  Deallocate(posrec)
316 
317  End Subroutine findbridge
