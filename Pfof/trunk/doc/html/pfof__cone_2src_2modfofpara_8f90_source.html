572  Integer(kind=4), dimension(:), allocatable :: adlast ! index of last particle "added" to a cube

574  Real(kind=4) :: xx, yy, zz ! temporary position
575  Real(kind=4) :: xmin,ymin,zmin ! minimum x,y and z of the subdomain
576

578

580 
581  pile = 0
582  adfirst = 0
583  adlast = 0
584  npcase = 0
585 
586  ! loop over the particles
587  Do ipart = 1,np
588  ! positions scaled to the resolution, i.e. between 0 and 512 for a 512^3 simulation for example
589  xx = (x(1,ipart)-xmin)*t*edgem1
590  yy = (x(2,ipart)-ymin)*t*edgem1
591  zz = (x(3,ipart)-zmin)*t*edgem1
592  ! coordinates of the cube containing the particle: this cube is always in the subdomain of the process
593  ix = int(xx)
594  iy = int(yy)
595  iz = int(zz)
596 
597  ! there may be some rounding problem because we switch from double precision to simple precision positions
598  ! so in case we have som positions = 1.0 we place the particle in the "last" cube
599  If(ix == n) ix = ix-1
600  If(iy == n) iy = iy-1
601  If(iz == n) iz = iz-1
602 
603  ! index of the cube
604  icase = 1 + ix + iy*n + iz*n*n
605  ! is there a bug somewhere with the index?
606  If(icase>ngpp .or. icase<=0) Then
607  print *,'Prob. pour une particule:',icase, ngpp,t, ix, iy, iz, n, xx, yy, zz,x(:,ipart), xmin, ymin, zmin
608  Call emergencystop('Probleme in initialization of friends of friends ',2)
609  End If
610  ! there is one more particle in this cube
611  npcase(icase) = npcase(icase) + 1
612  ! if this particle is the first added to the cube then
613  If(adfirst(icase) == 0) Then
614  ! we set the adfirst array to the particle index
615  adfirst(icase) = ipart
616  ! and the adlast as well because it's the last particle added a this time
617  adlast(icase) = ipart
618  Else
619  ! else the stack points from the previous particle to this one
620  pile(adlast(icase)) = ipart
621  ! and we set the adlast array
622  adlast(icase) = ipart
623  End If
624 
625  End Do
626 
627  Deallocate(adlast)
628
