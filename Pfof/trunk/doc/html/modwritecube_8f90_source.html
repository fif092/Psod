291  Subroutine h5writesortedcube()
292 
293  Use modsortinterf
294 
295  Implicit none
296 
297  Character(len=400) :: filecube
298  Character(len=5) :: pid_char
299  Character(len=8) :: charic
300 
301  Character(len=H5STRLEN) :: dsetname ! Dataset name
302  Character(len=H5STRLEN) :: aname ! Attribute name
303  Character(len=H5STRLEN) :: adata
304  Character(len=H5STRLEN) :: groupname
305  Character(len=H5STRLEN) :: codename
306 
307  Integer(hid_t) :: file_id ! File identifier
308  Integer(hid_t) :: gr_id
309  Integer(hid_t) :: gr_data_id
310 
311  Real(kind=4), dimension(6) :: boundaries
312 
313  Integer(kind=4) :: ic, ix, iy, iz, ip, deb, fin, nc, ncdim, deltam1
314  Integer(kind=4), dimension(:), allocatable :: npartcube
315  Integer(kind=PRI), dimension(:), allocatable :: ictable
316  Logical(kind=4) :: fileexist, fileopened
317  Integer(kind=8) :: npart8
318  Logical(kind=4) :: islast
319 
320 #ifdef DEBUG
321  print *,"Enter h5writesortedcube on process ",procid
322 #endif
323 
324  ! each FoF cube is divided into nc=512=8x8x8 groups, with ncdim=8
325  ncdim = 8
326  nc = ncdim*ncdim*ncdim
327  Allocate(ictable(mynpart))
328  Allocate(npartcube(nc))
329  npartcube = 0
330  ! Ramses coarse grid is composed of nres^3 cells
331  ! on a process: nres^3 / procNB cells
332  ! => there is (nres/(ncdim*dims(1))^3 coarse cells in each group
333  ! the size of a group is nres/(ncdim*dims(1))*(1/nres) where 1/nres is the size of 1 Ramses coarse cell
334  ! => delta = 1/(ncdim*dims(1)) => 1/delta = ncdim*dims(1)
335  deltam1 = ncdim*dims(1)
336 
337  ! We compute the "group" index of each particle and the number of particle in each group
338  Do ip=1, mynpart
339  ix = int((pos(1,ip) - xmin)*deltam1 + 1)
340  iy = int((pos(2,ip) - ymin)*deltam1 + 1)
341  iz = int((pos(3,ip) - zmin)*deltam1 + 1)
342  ! rounding issue
343  If(ix>ncdim) ix=ncdim
344  If(iy>ncdim) iy=ncdim
345  If(iz>ncdim) iz=ncdim
346  ic = ix + (iy-1)*ncdim + (iz-1)*ncdim*ncdim
347  ictable(ip) = ic
348  npartcube(ic) = npartcube(ic) + 1
349  End Do
350 
351  ! We sort the particles along their group id
352  If(param%do_read_potential .and. param%do_read_gravitational_field) Then
353  Call heapsort(mynpart,ictable,pos,vel,for,pot,id)
354  Else If (param%do_read_potential .and. .not. param%do_read_gravitational_field) Then
355  Call heapsort(mynpart,ictable,pos,vel,pot,id)
356  Else If (param%do_read_gravitational_field .and. .not. param%do_read_potential) Then
357  Call heapsort(mynpart,ictable,pos,vel,for,id)
358  Else
359  Call heapsort(mynpart,ictable,pos,vel,id)
360  End If
361 
362  ! We open a sortedcube file and write the groups into it
363  Write(pid_char(1:5),'(I5.5)') procid
364  filecube = 'pfof_cube_snap_part_data_'//trim(param%simulation_name)//'_'//pid_char//'.h5'
365 
366  Inquire(file=filecube,exist=fileexist, opened=fileopened)
367  If(fileopened) Then
368  print *,'Error on process ',procid,' : File ',filecube,' already opened'
369  End If
370 
371  ! create the hdf5 file
372  Call hdf5_create_file(filecube, file_id)
373 
374  codename='pfof_snap'
375  npart8 = 2**(inforamses%levelmin)
376  npart8 = npart8**3
377  Call h5write_meta_common(file_id, codename, npart8, procid)
378  Call h5write_meta_pfof_parameter(file_id, param)
379  islast = .false.
380  Call h5write_meta_info_ramses(file_id, inforamses, islast)
381 
382  ! open the root group
383  groupname = 'metadata'
384  Call hdf5_open_group(file_id, groupname, gr_id)
385 
386  ! Write type as attribute
387  aname = 'file_type'
388  adata = 'sortedcube'
389  Call hdf5_write_attr(gr_id, aname, adata)
390 
391  ! Write the number of particles as an attribute
392  dsetname = 'npart_file'
393  npart8 = int(mynpart, kind=8)
394  Call hdf5_write_data(gr_id, dsetname, npart8)
395 
396  ! Write the process ID as an attribute
397  aname = 'procID'
398  Call hdf5_write_attr(gr_id, aname, procid)
399 
400  aname = 'nfile'
401  Call hdf5_write_attr(gr_id, aname, procnb)
402 
403  ! Write the boundaries of the cube as an attribute
404  boundaries(1) = xmin
405  boundaries(2) = xmax
406  boundaries(3) = ymin
407  boundaries(4) = ymax
408  boundaries(5) = zmin
409  boundaries(6) = zmax
410  aname = 'boundaries'
411  Call hdf5_write_attr(gr_id, aname, 6, boundaries)
412 
413  aname='ngroup'
414  Call hdf5_write_attr(gr_id, aname, nc)
415 
416  aname='1/groupsize'
417  Call hdf5_write_attr(gr_id, aname, deltam1)
418 
419  dsetname = 'npart_grp_array'
420  Call hdf5_write_data(gr_id, dsetname, nc, npartcube)
421 
422  Call hdf5_close_group(gr_id)
423 
424  groupname = 'data'
425  Call hdf5_create_group(file_id, groupname, gr_data_id)
426 
427  deb = 1
428  ! For each non empty group we create an HDF5 group and write dataset into it
429  Do ic = 1, nc
430  If(npartcube(ic) /= 0) Then
431  Write(charic(1:8),'(I8.8)') ic
432  groupname = 'group'//charic
433 
434  Call hdf5_create_group(gr_data_id,groupname,gr_id)
435 
436  fin = deb + npartcube(ic) - 1
437 
438  ! Write the position of the particles
439  dsetname='position_part'
440  Call hdf5_write_data(gr_id, dsetname, 3, npartcube(ic), pos(:,deb:fin))
441 
442  ! Write the velocity of the particles
443  dsetname='velocity_part'
444  Call hdf5_write_data(gr_id, dsetname, 3, npartcube(ic), vel(:,deb:fin))
445 
446  ! Write the ID of the particles
447  dsetname='identity_part'
448  Call hdf5_write_data(gr_id, dsetname, npartcube(ic), id(deb:fin))
449 
450  ! Write potential if it is used
451  If(param%do_read_potential) Then
452  dsetname = 'potential_part'
453  Call hdf5_write_data(gr_id, dsetname, npartcube(ic), pot(deb:fin))
454  End If
455 
456  ! Write force if it is used
457  If(param%do_read_gravitational_field) Then
458  dsetname = 'gravitational_field_part'
459  Call hdf5_write_data(gr_id, dsetname, 3, npartcube(ic), for(:,deb:fin))
460  End If
461 
462  Call hdf5_close_group(gr_id)
463  deb = fin + 1
464 
465  End If
466  End Do
467 
468  Call hdf5_close_group(gr_data_id)
469 
470  Call hdf5_close_file(file_id)
471 
472 #ifdef DEBUG
473  print *,'Exit h5writesortedcube on process ',procid
474 #endif
475 
476  End Subroutine h5writesortedcube
