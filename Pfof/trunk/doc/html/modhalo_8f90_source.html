404  halocompos(:,h) = halopartpos(:,fi)
405  halocomvel(:,h) = halopartvel(:,fi)
406  Do i = fi+1, li
407  delta(:) = halocompos(:,h) / halom - halopartpos(:,i)
408  If(periodic) Then
409  Do j = 1, 3
410  If(abs(delta(j)) > 0.5d0) Then
411  If(delta(j) > 0d0) halocompos(j,h) = halocompos(j,h) + 1.0d0
412  If(delta(j) < 0d0) halocompos(j,h) = halocompos(j,h) - 1.0d0
413  End If
414  End Do
415  End If

417  halocompos(:,h) = halocompos(:,h) + halopartpos(:,i)
418  halocomvel(:,h) = halocomvel(:,h) + halopartvel(:,i)
419  End Do
420  fi = li + 1
421 
422  End Do
423 
424  Do h = 1, halonb
425  halocompos(:,h) = halocompos(:,h) / halomass(h)
426  halocomvel(:,h) = halocomvel(:,h) / halomass(h)
427  If(periodic) Then
428  Do j = 1, 3
429  If(halocompos(j,h) > 1.0d0) halocompos(j,h) = halocompos(j,h) - 1.0d0
430  If(halocompos(j,h) < 0.0d0) halocompos(j,h) = halocompos(j,h) + 1.0d0
431  End Do
432  End If

473  End If

475  If(d2 > rmax2) rmax2 = d2
476  End Do
