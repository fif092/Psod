42  Subroutine setcommunicators()
43 
44  Use modparameters
45  Implicit None
46 
47  ! Creation of a grid of processes.
48  ! The grid is a cube, there are procNB**(1/3) processes in each dimension.
49  ! The grid is periodic in each dimension.
50  dims = int(procnb**(1./3.))
51  periods = .true.
52 
53  Call mpi_cart_create(mpi_comm_world,3,dims,periods,.true.,mpicube,mpierr)
54 
55  ! Process ID and number
56  Call mpi_comm_rank (mpicube, procid, mpierr)
57  Call mpi_cart_coords(mpicube, procid, 3, cubecoord, mpierr)
58 
59  ! Definition des neighbourss pour les 6 directions avec des conditions au bord periodiques
60  ! Intialization of the 'neighboor processes id'
61  ! Neighboor : 1 = backward
62  ! 2 = forward
63  ! 3 = left
64  ! 4 = right
65  ! 5 = down
66  ! 6 = up
67 
68  Call mpi_cart_shift(mpicube,0,1,neighbours(1),neighbours(2),mpierr)
69  Call mpi_cart_shift(mpicube,1,1,neighbours(3),neighbours(4),mpierr)
70  Call mpi_cart_shift(mpicube,2,1,neighbours(5),neighbours(6),mpierr)
71 
72 
73  If(param%gatherwrite_factor > 1) Then
74  If(procid==0) Then
75  print *, '*** Creating specific communicator for gathered cube output ***'
76  End If
77 
78  commcolorwrite = cubecoord(1)/param%gatherwrite_factor + &
79  (cubecoord(2)/param%gatherwrite_factor) * (dims(1)/param%gatherwrite_factor) &
80  + (cubecoord(3)/param%gatherwrite_factor) * (dims(1)/param%gatherwrite_factor)*&
81  (dims(1)/param%gatherwrite_factor)
82 
83  Call mpi_comm_split(mpicube, commcolorwrite, procid, mpisubcubewrite, mpierr)
84 
85  Call mpi_comm_rank(mpisubcubewrite, scprocidwrite, mpierr)
86  Call mpi_comm_size(mpisubcubewrite, scprocnbwrite, mpierr)
87  End If
88 
89  If(param%gatherread_factor > 1) Then
90 
91  If(procid==0) Then
92  print *, '*** Creating specific communicator for gathered cube input ***'
93  End If
94 
95  commcolorread = cubecoord(1)/param%gatherread_factor + &
96  (cubecoord(2)/param%gatherread_factor) * (dims(1)/param%gatherread_factor) &
97  + (cubecoord(3)/param%gatherread_factor) * (dims(1)/param%gatherread_factor)* &
98  (dims(1)/param%gatherread_factor)
99 
100  Call mpi_comm_split(mpicube, commcolorread, procid, mpisubcuberead, mpierr)
101 
102  Call mpi_comm_rank(mpisubcuberead, scprocidread, mpierr)
103  Call mpi_comm_size(mpisubcuberead, scprocnbread, mpierr)
104 
105  End If
