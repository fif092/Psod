503  If(ix == n) ix = ix-1
504  If(iy == n) iy = iy-1
505  If(iz == n) iz = iz-1

508  icase = 1 + ix + iy*n + iz*n*n
509  ! is there a bug somewhere with the index?
510  If(icase>ngpp .or. icase<=0) Then
511  Call emergencystop('Probleme in initialization of friends of friends ',12)
512  End If
513  ! there is one more particle in this cube
514  npcase(icase) = npcase(icase) + 1

516  If(adfirst(icase) == 0) Then
517  ! we set the adfirst array to the particle index
518  adfirst(icase) = ipart
